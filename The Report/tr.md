    SELECT NAME, GRADE, MARKS AS MARK
    FROM STUDENTS
    LEFT JOIN GRADES ON STUDENTS.MARKS >= GRADES.MIN_MARK AND STUDENTS.MARKS <= GRADES.MAX_MARK
    WHERE GRADE >= 8
    ORDER BY GRADE DESC, NAME
    ;
    SELECT 'NULL', GRADE, MARKS AS MARK
    FROM STUDENTS
    LEFT JOIN GRADES ON STUDENTS.MARKS >= GRADES.MIN_MARK AND STUDENTS.MARKS <= GRADES.MAX_MARK
    WHERE GRADE < 8
    ORDER BY GRADE DESC, MARK
    ;
